cargar_fichero_r <- function(ruta, separador = ",") {
  df <- read.csv(ruta, sep = separador) |> 
    select(any_of(c("RB010", "RB030", "RB080", 
                    "RB090", "RB220", "RB230", "RB240",
                    "RB062", "RB063", "RB064")))
  return(df)
}

Unión Ficheros Longitudinales

Como algunas variables venían en algún archivo como caracter y en otras como numérica he convertido todas en caracter primero para que se unieran bien y luego ya cuando necesite hacer cálculos las cambio a numérica. Tammbién he eliminado duplicados porque como se solapan los años para que no haya más de una persona en el mismo año:

# 1. Estandarizar tipos
fichero_r_base2004 <- fichero_r_base2004 |> mutate(across(everything(), as.character))
fichero_r_base2013_csv <- fichero_r_base2013_csv  |>  mutate(across(everything(), as.character))
fichero_r_nuevo_csv <- fichero_r_nuevo_csv  |>  mutate(across(everything(), as.character))
fichero_r_rdata <- fichero_r_rdata  |>  mutate(across(everything(), as.character))

# 2. Unir todos
fichero_r_total <- bind_rows(
  fichero_r_base2004,
  fichero_r_base2013_csv,
  fichero_r_nuevo_csv,
  fichero_r_rdata

# 3. Eliminar duplicados  
fichero_r_total <- fichero_r_total  |>

Variable Emancipado y Edad

fichero_r_total <- fichero_r_total |> 
  mutate(
    RB010 = as.numeric(RB010),
    RB030 = as.numeric(RB030),
    RB080 = as.numeric(RB080),
    RB220 = as.numeric(RB220),
    RB230 = as.numeric(RB230),
    edad  = RB010 - RB080,
    emancipado = case_when(
      (is.na(RB220) | RB220 == 0) &
      (is.na(RB230) | RB230 == 0) ~ 1,
      TRUE ~ 0
    )
  )  |> 
  filter(edad >= 18 & edad <= 38)

Detección de transiciones y edad de emancipación

Uso gap = 1 para quedarme con las transiciones entre años consecutivos

transiciones_total <- fichero_r_total |> 
  arrange(RB030, RB010) |> 
  group_by(RB030) |> 
  mutate(
    emancipado_anterior = lag(emancipado),
    año_anterior        = lag(RB010),
    gap                 = RB010 - año_anterior,
    se_emancipa = case_when(
      emancipado == 1 & emancipado_anterior == 0 & gap == 1 ~ 1,
      TRUE ~ 0
    )
  )  |> 
  filter(!is.na(emancipado_anterior), gap == 1)  |> 
  ungroup()
# Cuántas transiciones hay por año
transiciones_total  |> 
  group_by(RB010) |> 
  summarise(
    n_total       = n(),
    n_se_emancipa = sum(se_emancipa == 1),
    pct_transicion = round(mean(se_emancipa == 1) * 100, 2)
  )  |> 
  print(n = 25)

Gráfico Tasa de Transición por año

transiciones_total |>
  group_by(RB010) |>
  summarise(
    n_total        = n(),
    n_se_emancipa  = sum(se_emancipa == 1),
    pct_transicion = round(mean(se_emancipa == 1) * 100, 2)

  filter(RB010 <= 2021, n_total >= 100) |>   # excluir años con pocas obs
  ggplot(aes(x = RB010, y = pct_transicion)) +
  geom_line(color = "#2d6a4f", linewidth = 1.2) +
  geom_point(color = "#2d6a4f", size = 3) +
  geom_smooth(method = "lm", se = TRUE, color = "#d73027",
              linetype = "dashed", linewidth = 0.8) +
  geom_text(aes(label = paste0(pct_transicion, "%")),
            vjust = -1, size = 3.5) +
  scale_x_continuous(breaks = seq(2005, 2021, by = 2)) +
  scale_y_continuous(limits = c(4, 12),
                     labels = function(x) paste0(x, "%")) +
  labs(
    title    = "Tasa de transición a la emancipación en España",
    subtitle = "Proporción de jóvenes 18-38 años que se emancipan cada año",
    x        = "Año",
    y        = "Tasa de transición (%)"
  ) +
  theme_minimal(base_size = 13)
