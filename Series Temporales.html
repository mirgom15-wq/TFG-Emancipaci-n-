resultados_csv_antiguo <- map_dfr(2004:2015, function(año) {
  ruta_r <- paste0(ruta_base, año, "/esudb", substr(año, 3, 4), "r.csv")
  ruta_d <- paste0(ruta_base, año, "/esudb", substr(año, 3, 4), "d.csv")
  
  datos_r <- read_csv(ruta_r, show_col_types = FALSE)
  datos_d <- read_csv(ruta_d, show_col_types = FALSE)
  
  datos_d <- datos_d |> rename(id_hogar = DB030)
  datos_r <- datos_r |> mutate(id_hogar = RB030 %/% 100)
  
  datos <- datos_r |>
    left_join(datos_d |> select(id_hogar, DB090), by = "id_hogar") |>
    mutate(edad = año - RB080) |>
    filter(edad >= 18 & edad <= 38) |>
    mutate(emancipado = ifelse(is.na(RB220) & is.na(RB230), 1, 0))
  
  tasa <- datos |>
    group_by(emancipado) |>
    summarise(total = sum(DB090, na.rm = TRUE), .groups = "drop") |>
    mutate(porcentaje = round(total / sum(total) * 100, 2)) |>
    filter(emancipado == 1) |>
    pull(porcentaje)
  
  edad_media <- datos |>
    filter(emancipado == 1) |>
    summarise(edad_media = weighted.mean(edad, DB090, na.rm = TRUE)) |>
    pull(edad_media) |>
    round(1)
  
  data.frame(año = año, tasa_emancipacion = tasa, edad_media_emancipados = edad_media)
})
